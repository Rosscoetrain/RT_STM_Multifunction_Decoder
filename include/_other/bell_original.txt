#include <Arduino.h>


class RailroadBell {
private:
    uint8_t _pin;
    uint32_t _freq;
    uint32_t _interval;
    uint32_t _strikeDuration;
    unsigned long _lastStrikeTime;
    bool _isRinging;

public:
    RailroadBell(uint8_t pin, uint32_t freq, uint32_t interval) 
        : _pin(pin), _freq(freq), _interval(interval), 
          _strikeDuration(120), _lastStrikeTime(0), _isRinging(false) {}

    void begin() {
        pinMode(_pin, OUTPUT);
        _lastStrikeTime = millis();
    }

    void update() {
        unsigned long currentTime = millis();

        // Start a new strike
        if (!_isRinging && (currentTime - _lastStrikeTime >= _interval)) {
            tone(_pin, _freq);
            _isRinging = true;
            _lastStrikeTime = currentTime;
        }

        // End the current strike (decay phase)
        if (_isRinging && (currentTime - _lastStrikeTime >= _strikeDuration)) {
            noTone(_pin);
            _isRinging = false;
        }
    }
};


// PA7 pin, 850Hz, 1000ms strike interval
RailroadBell bell(PA7, 850, 1000);

void setup() {
    bell.begin();
}

void loop() {
    // This now runs at full speed without stopping
    bell.update();
}

#include <Arduino.h>

class RailroadBell {
private:
    uint8_t _pin;
    uint32_t _freq;
    uint32_t _interval;
    uint32_t _strikeDuration;
    unsigned long _lastStrikeTime;
    bool _isRinging;
    bool _enabled; // The ON/OFF state

public:
    RailroadBell(uint8_t pin, uint32_t freq, uint32_t interval) 
        : _pin(pin), _freq(freq), _interval(interval), 
          _strikeDuration(120), _lastStrikeTime(0), 
          _isRinging(false), _enabled(false) {}

    void begin() {
        pinMode(_pin, OUTPUT);
        noTone(_pin);
    }

    // New control functions
    void setEnabled(bool state) {
        _enabled = state;
        if (!_enabled) {
            noTone(_pin); // Force silence when turned off
            _isRinging = false;
        }
    }

    bool isEnabled() const { return _enabled; }

    void update() {
        if (!_enabled) return;

        unsigned long currentTime = millis();

        // Start strike
        if (!_isRinging && (currentTime - _lastStrikeTime >= _interval)) {
            tone(_pin, _freq);
            _isRinging = true;
            _lastStrikeTime = currentTime;
        }

        // End strike
        if (_isRinging && (currentTime - _lastStrikeTime >= _strikeDuration)) {
            noTone(_pin);
            _isRinging = false;
        }
    }
};

// PA7 pin, 850Hz, 1000ms interval
RailroadBell bell(PA7, 850, 1000);

void setup() {
    bell.begin();
    
    // Example: Turn the bell ON initially
    bell.setEnabled(true);
}

void loop() {
    bell.update();

    // Example logic: Turn off after 10 seconds
    if (millis() > 10000) {
        bell.setEnabled(false);
    }
}
