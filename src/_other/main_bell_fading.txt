#include <Arduino.h>

class RailroadBell {
private:
    uint8_t _pin;
    uint32_t _baseFreq;
    uint32_t _interval;
    uint32_t _strikeDuration;
    unsigned long _lastStrikeTime;
    bool _isRinging;
    bool _enabled;

public:
    RailroadBell(uint8_t pin, uint32_t baseFreq, uint32_t interval) 
        : _pin(pin), _baseFreq(baseFreq), _interval(interval), 
          _strikeDuration(400), _lastStrikeTime(0), // Longer duration for natural fade
          _isRinging(false), _enabled(false) {}

    void begin() {
        pinMode(_pin, OUTPUT);
        noTone(_pin);
    }

    void setEnabled(bool state) {
        _enabled = state;
        if (!_enabled) {
            noTone(_pin); 
            _isRinging = false;
        }
    }

    bool isEnabled() const { return _enabled; }

    void update() {
        if (!_enabled) return;

        unsigned long currentTime = millis();

        // Trigger a new strike
        if (!_isRinging && (currentTime - _lastStrikeTime >= _interval)) {
            _isRinging = true;
            _lastStrikeTime = currentTime;
        }

        // Handle the ringing decay animation
        if (_isRinging) {
            unsigned long elapsed = currentTime - _lastStrikeTime;

            if (elapsed < _strikeDuration) {
                // Non-linear frequency drop to simulate resonance decay
                // Fast drop at first, then slow trailing vibration
                float progress = (float)elapsed / _strikeDuration;
                float decayFactor = 1.0 - (progress * progress); 
                
                // Add a second harmonic (overtone) by shifting the pitch down rapidly
                uint32_t currentFreq = _baseFreq + (uint32_t)(400.0 * decayFactor);
                
                tone(_pin, currentFreq);
            } else {
                noTone(_pin);
                _isRinging = false;
            }
        }
    }
};

// Initialise on PA2, 550Hz base, 1000ms strike interval
// 500Hz-600Hz is standard for heavy cast-iron mechanical bells
RailroadBell bell(PA2, 550, 1000);

void setup() {
    bell.begin();
    bell.setEnabled(true);
}

void loop() {
    bell.update();
}
